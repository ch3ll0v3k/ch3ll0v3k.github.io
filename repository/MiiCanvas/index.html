    onLoad(function() {
        
        let MiiCan = new MiiCanvas({canvasId:'my-canvas-0', W:600, H:300});

        MiiCan.setBg( '#444' );

        let x = 0;
        let y = 0;
        let val = 0;
        let radius = 100;

        MiiCan.setFrameRate( 15 );
        MiiCan.onFrame(function( frame, mSecPast, secPast ){

            MiiCan.clear();
            MiiCan.clearInfoTextIndex();

            MiiCan.drawRect( 300, 200, 200, 30, '#F00' );
            MiiCan.drawLine( 30, 100, 200, 160, '#FF0' );
            MiiCan.drawDot( 300, 100, 40, '#F95' );
            MiiCan.drawDotField( 300, 100, 60, '#F00' );
            MiiCan.infoText(' # MiiCanvas.js: [0]', '#eee' );
            MiiCan.infoText(' # MiiCanvas.js: [1]', '#eee' );
            MiiCan.infoText(' # MiiCanvas.js: [2]', '#eee' );
            MiiCan.infoText(' # MiiCanvas.js: [3]', '#eee' );
            MiiCan.drawText(' # MiiCanvas.js', 350, 50, '#0F0' );

            val += 0.1;

            MiiCan.drawLine( 0, y, MiiCan.W, y, '#0F0', 1 );
            MiiCan.drawLine( x, 0, x, MiiCan.H, '#0F0', 1 );

            x = MiiCan.CW - ( Math.cos( val ) * radius );
            y = MiiCan.CW - ( Math.sin( val ) * radius );

            MiiCan.drawDot( x, y, 10, '#F95' );


        });

        MiiCan.startAnimation();

        MiiCan.onMouseDown(function(e, x, y){
            console.log( ' #MiiCan.onMouseDown: ', {x,y, isMouseDown} );
        });

        MiiCan.onMouseUp(function(e, x, y){
            console.log( ' #MiiCan.onMouseUp: ', {x,y, isMouseDown} );
        });

        MiiCan.onMouseMove(function(e, x, y, isMouseDown){
            console.log( ' #MiiCan.onMouseMove: ', {x,y, isMouseDown} );
        });

    });
